To configure colorscheme
1. create new file .vimrc in home directory called .vimrc
2. copy+paste the following in:

syntax on                                                                                            
                                                                                                     
colorscheme openDyslexia                                                                             
                                                                                                     
highlight Normal ctermfg=white ctermbg=16                                                            
                                                                                                     
if $COLORTERM == 'gnome-terminal'                                                                    
        set t_Co=256                                                                                 
endif                                                                                                
                                                                                                     
if (&t_Co == 256)                                                                                    
  set t_ut=                                                                                          
endif
